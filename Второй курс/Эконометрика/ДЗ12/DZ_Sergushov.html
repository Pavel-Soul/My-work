# Создание индикатора для цензурирования
data['Age_Censored'] = 0
data.loc[data['Age'] < 18, 'Age_Censored'] = 1
data.loc[data['Age'] > 70, 'Age_Censored'] = 2

data.loc[data['Age_Censored'] > 0, 'Age'] = np.nan

data.drop(['CustomerID', 'Genre', 'Age_Censored'], axis=1, inplace=True)
data.dropna(inplace=True)

X = data[['Annual Income (k$)', 'Spending Score (1-100)']]
Y = data['Age']
X_with_const = sm.add_constant(X)

left_censoring = (Y.isnull()) * 1
right_censoring = (Y.isnull()) * 1

Y_censored = Y.copy()
Y_censored[left_censoring == 1] = np.nan
Y_censored[right_censoring == 1] = np.nan

model = sm.OLS(Y_censored, X_with_const)
fit = model.fit()

predicted = fit.predict()

Y_predicted = Y.copy()
Y_predicted[left_censoring == 1] = predicted[left_censoring == 1]
Y_predicted[right_censoring == 1] = predicted[right_censoring == 1]

print(fit.summary())

from py4etrics import heckit
import matplotlib.pyplot as plt
import warnings
from statsmodels.tools.sm_exceptions import PerfectSeparationWarning

warnings.simplefilter("ignore", PerfectSeparationWarning)

def draw_pred(y, y_pred):
    fig, ax = plt.subplots()
    ax.scatter(y, y_pred)
    ax.plot([min(y), max(y)], [min(y_pred), max(y_pred)], 'r')

data = pd.read_csv("Mall_Customers.csv")

x = data[['Annual Income (k$)', 'Spending Score (1-100)']]

x = sm.add_constant(x)

model = heckit.Heckit(y, x, x).fit()

y_pred = model.predict(x)
