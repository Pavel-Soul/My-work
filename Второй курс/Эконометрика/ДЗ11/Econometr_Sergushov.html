y = df['Close'].values

def foster_stuart_trend(y):
    n = len(y)
    k = np.zeros(n)
    l = np.zeros(n)

    for t in range(1, n):
        if y[t] > max(y[:t]):
            k[t] = 1
        if y[t] < min(y[:t]):
            l[t] = 1

    s = np.sum(k[2:]) + np.sum(l[2:])
    d = np.sum(k[2:]) - np.sum(l[2:])

    mu = (1.693872 * np.log(n) - 0.299015) / (1 - 0.035092 * np.log(n) + 0.002705 * np.log(n) ** 2)
    sigma1 = np.sqrt(2 * np.log(n) - 3.4253)
    sigma2 = np.sqrt(2 * np.log(n) - 0.8456)

    t_sr = np.abs(s - mu) / sigma1
    t_d = np.abs(d) / sigma2

    t_crit = np.sqrt(2 * np.log(n))

    return t_crit, t_sr, t_d
