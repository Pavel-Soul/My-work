$P(t) = \phi \cdot K(t)$

$\pi_0(t) = [P(t) - C(t) - L(t)] \cdot (1-\tau_1)-W(t) - EW(t) - A(t)$

$\pi(t) = (1 - \tau_2) \cdot \pi_0(t)$

$\frac {dK(t)}{dt} = \alpha \cdot \pi(t) + S(t)$

$\alpha \in (0,1), t \in [0,T]$

$C(t) = c_1, L(t) = c_2, W(t) = c_3, EW(t) = c_4, A(t) = c_5$

$\frac {dK(t)} {dt} = \alpha \cdot \pi(t) + S(t) = \alpha \cdot [(1 - \tau_2 \cdot\pi_0(t)]+ S(t) = \alpha \cdot[(1 - \tau_2){[\phi \cdot K(t) - (c_1+c_2)]\cdot(1-\tau_1)-(c_3+c_4+c_5)}] + S(t) = \alpha \cdot \phi \cdot (1-\tau_1)\cdot (1-\tau_2)\cdot K(t) + S(t) - \alpha \cdot(1-\tau_2)\cdot[(1-\tau_1)\cdot (c_1 + c_2) + (c_3+c_4+c_5)]$

$a = \alpha \cdot \phi \cdot (1-\tau_1) \cdot (1 - \tau_2)$

$b = -\alpha \cdot (1 - \tau_2) \cdot [(1-\tau_1)\cdot (c_1 + c_2) + (c_3+c_4+c_5)]$

$\frac {dK(t)}{dt} = a \cdot K(t) + S(t) + b$
