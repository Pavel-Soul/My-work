import numpy as np
import matplotlib.pyplot as plt

def utility_function(x1, x2):
    return np.sqrt(x1) + np.sqrt(x2)

utility_levels = [2, 4, 6, 8, 10]

x1_values = np.linspace(0, 10, 400)

for utility_level in utility_levels:
    x2_values = [(utility_level - np.sqrt(x1))**2 for x1 in x1_values]
    plt.plot(x1_values, x2_values, label=f'U = {utility_level}')

plt.xlabel('x1')
plt.ylabel('x2')
plt.title(' ')
plt.grid(True)
plt.legend()
plt.show()

x1, x2 = sp.symbols('x1 x2')

u = sp.sqrt(x1) + sp.sqrt(x2)

mu1 = sp.diff(u, x1)
mu2 = sp.diff(u, x2)

print("   1 (MU1):", mu1)
print("   2 (MU2):", mu2)

mrs = mu1 / mu2

print("   (MRS):", mrs)

elasticity_x1 = sp.diff(mu1 * x1, x1) / mu1 / x1
elasticity_x2 = sp.diff(mu2 * x2, x2) / mu2 / x2

print("     1:", elasticity_x1)
print("     2:", elasticity_x2)

   1 (MU1): 1/(2*sqrt(x1))
   2 (MU2): 1/(2*sqrt(x2))
   (MRS): sqrt(x2)/sqrt(x1)
     1: 1/(2*x1)
     2: 1/(2*x2)

import numpy as np
import matplotlib.pyplot as plt

def utility_function(x1, x2):
    return 9 * np.log(x1 + 4) + 4 * np.log(x2 + 9)

utility_levels = [10, 20, 30, 40, 50]

x1_values = np.linspace(0.1, 50, 400)
x2_values = np.linspace(0.1, 50, 400)

for utility_level in utility_levels:
    x1_values_for_level = [x1 for x1 in x1_values]
    x2_values_for_level = [(np.exp((utility_level - 9 * np.log(x1 + 4)) / 4) - 9) for x1 in x1_values]
    plt.plot(x1_values_for_level, x2_values_for_level, label=f'U = {utility_level}')

plt.xlabel('x1')
plt.ylabel('x2')
plt.title(' ')
plt.grid(True)
plt.legend()
plt.show()
